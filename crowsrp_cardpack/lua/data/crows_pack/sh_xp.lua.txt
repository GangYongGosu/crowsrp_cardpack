-- Called when player spawns
hook.Add("PlayerInitialSpawn", "CROWS_TCG_InitXP", function(ply)
    ply:SetNWInt("CROWS_XP", 0)
    ply:SetNWInt("CROWS_Level", 1)
end)

-- XP Required per Level
local function XPForLevel(lvl)
    return math.floor(100 + (lvl ^ 1.5) * 50)
end
