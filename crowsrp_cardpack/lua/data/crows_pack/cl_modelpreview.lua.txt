function CrowsPack_OpenModelPreview(modelPath)
    local preview = vgui.Create("DFrame")
    preview:SetSize(500, 500)
    preview:Center()
    preview:SetTitle("Preview")
    preview:MakePopup()

    local modelPanel = vgui.Create("DModelPanel", preview)
    modelPanel:Dock(FILL)
    modelPanel:SetModel(modelPath)

    function modelPanel:LayoutEntity(ent)
        ent:SetAngles(Angle(0, RealTime() * 20, 0))
    end

    modelPanel:SetFOV(40)
end
