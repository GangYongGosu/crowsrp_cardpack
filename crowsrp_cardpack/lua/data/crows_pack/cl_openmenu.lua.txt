local function RevealCards(cardDataList)
    local frame = vgui.Create("DFrame")
    frame:SetSize(800, 500)
    frame:Center()
    frame:SetTitle("Pack Opening")
    frame:MakePopup()

    local cards = {}

    for i = 1, #cardDataList do
        local card = vgui.Create("CrowsPackCard", frame)
        card:SetPos(30 + ((i-1)%5)*150, 60 + math.floor((i-1)/5)*200)
        card:SetCardInfo(cardDataList[i].rarity, cardDataList[i])
        table.insert(cards, card)
    end

    timer.Create("FlipCardsAnim", 0.6, #cards, function()
        local i = timer.RepsLeft("FlipCardsAnim")
        if cards[i] then
            cards[i]:Flip()
            surface.PlaySound("crows_pack/rare_reveal.wav")
        end
    end)
end
