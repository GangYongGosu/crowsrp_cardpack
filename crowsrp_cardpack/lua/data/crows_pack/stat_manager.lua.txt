function CrowsRP.GenerateCardStats(rarity)
    local durability = math.random(50, 500)
    local rank = CrowsRP.GetDurabilityRank(durability)

    return {
        Durability = durability,
        Weight = math.random(1, 10),
        Armor = math.random(0, 100),
        Health = math.random(0, 100),
        DamageBoost = math.random(0, 25),
        Accuracy = math.random(0, 100),
        FireRate = math.random(1, 10),
        PassiveEffect = CrowsRP.GetRandomPassive(),
        Rank = rank
    }
end
